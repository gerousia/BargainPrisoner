## GM1 scripted_triggers

is_under_ongoing_interaction = {
	custom_description = {															
		text = "is_under_ongoing_interaction_trigger"
		has_character_flag = under_transfer_prisoner_interaction
	}
}

has_recently_undergone_interaction = {
	custom_description = {
		text = "has_recently_undergone_interaction_trigger"
		has_character_flag = recently_prison_transferred
	}
}

is_at_imprisoner_capital = {
	trigger_if = {
		limit = { exists = imprisoner }
		custom_description = {
			text = "is_at_imprisoner_capital_trigger"
			object = imprisoner
			is_at_location = imprisoner.capital_province
		}
	}	
}
